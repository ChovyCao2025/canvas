# 多级缓存 SDK Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将多级缓存模式抽离为独立 Maven 模块 `canvas-cache-sdk`，实现泛型 L1+L2+L3 三级缓存、Pub/Sub 跨实例 L1 失效广播、L1 续期（refreshAfterWrite 优先查 L2），并将 `CanvasConfigCache` 迁移至 SDK。

**Architecture:** 独立 Maven 模块（无业务依赖）→ Spring Boot AutoConfiguration → TieredCacheManager 管理 Pub/Sub → TieredCache\<K,V\> 泛型实现 → CanvasConfigCache 薄包装。

**Tech Stack:** Java 21, Caffeine, Spring Data Redis, Jackson, Spring Boot AutoConfiguration

```java
package org.chovy.cache;

import java.util.Optional;

/**
 * 多级缓存接口：L1(Caffeine) + L2(Redis) + L3(Loader)。
 *
 * @param <K> 缓存键类型（需可序列化为 String）
 * @param <V> 缓存值类型（需 Jackson 可序列化）
 */
public interface TieredCache<K, V> {

    /** 缓存名称，用于 Pub/Sub channel 隔离：tiered-cache:{name}:invalidate */
    String name();

    /**
     * L1 → L2 → Loader 三级查找，结果自动回填上级缓存。
     * @return Optional.empty() 当 Loader 返回 null 时
     */
    Optional<V> get(K key);

    /** get() + 未命中时抛 NoSuchElementException */
    default V getOrThrow(K key) {
        return get(key).orElseThrow(() ->
                new java.util.NoSuchElementException("Cache miss: " + name() + " key=" + key));
    }

    /** 写入 L1 + L2（外部主动 put，绕过 Loader） */
    void put(K key, V value);

    /**
     * 清除 L1 + L2，并通过 Redis Pub/Sub 广播给所有实例的 L1 Caffeine。
     */
    void invalidate(K key);

    /** 强制从 Loader 重新加载，更新 L1 + L2 */
    void refresh(K key);
}
```

import java.time.Duration;
import java.util.function.Function;

/** Fluent builder for TieredCache. */
@NoArgsConstructor(access = AccessLevel.PRIVATE)
public class TieredCacheBuilder<K, V> {

    private String name;
    private int l1MaxSize = 1000;
    private Duration l1RefreshAfterWrite = Duration.ofHours(1);
    private String l2KeyPrefix = "cache:";
    private Duration l2Ttl = Duration.ofHours(24);
    private Function<K, V> loader;
    private JavaType valueJavaType;
    private ObjectMapper objectMapper = new ObjectMapper();

    public static <K, V> TieredCacheBuilder<K, V> builder() {
        return new TieredCacheBuilder<>();
    }

    public TieredCacheBuilder<K, V> name(String name) {
        this.name = name; return this;
    }

    public TieredCacheBuilder<K, V> l1MaxSize(int size) {
        this.l1MaxSize = size; return this;
    }

    public TieredCacheBuilder<K, V> l1RefreshAfterWrite(Duration ttl) {
        this.l1RefreshAfterWrite = ttl; return this;
    }

    public TieredCacheBuilder<K, V> l2KeyPrefix(String prefix) {
        this.l2KeyPrefix = prefix; return this;
    }

    public TieredCacheBuilder<K, V> l2Ttl(Duration ttl) {
        this.l2Ttl = ttl; return this;
    }

    public TieredCacheBuilder<K, V> loader(Function<K, V> loader) {
        this.loader = loader; return this;
    }

    public TieredCacheBuilder<K, V> valueType(Class<V> clazz) {
        this.valueJavaType = objectMapper.constructType(clazz); return this;
    }

    public TieredCacheBuilder<K, V> valueType(com.fasterxml.jackson.core.type.TypeReference<V> ref) {
        this.valueJavaType = objectMapper.constructType(ref.getType()); return this;
    }

    public TieredCacheBuilder<K, V> objectMapper(ObjectMapper om) {
        this.objectMapper = om; return this;
    }

    /** 构建并注册到 TieredCacheManager（自动订阅失效广播）。 */
    public TieredCache<K, V> build(TieredCacheManager manager) {
        if (name == null || name.isBlank()) throw new IllegalStateException("name is required");
        if (loader == null)                 throw new IllegalStateException("loader is required");
        if (valueJavaType == null)          throw new IllegalStateException("valueType is required");

        TieredCacheImpl<K, V> cache = new TieredCacheImpl<>(
                name, l1MaxSize, l1RefreshAfterWrite,
                l2KeyPrefix, l2Ttl, loader, valueJavaType, objectMapper,
                manager.getRedis()
        );
        manager.register(cache);
        return cache;
    }
}

```java
package org.chovy.cache;

import com.fasterxml.jackson.databind.JavaType;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.github.benmanes.caffeine.cache.CacheLoader;
import com.github.benmanes.caffeine.cache.Caffeine;
import com.github.benmanes.caffeine.cache.LoadingCache;
import lombok.Getter;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.StringRedisTemplate;

import java.time.Duration;
import java.util.Optional;
import java.util.function.Function;

@Slf4j
public class TieredCacheImpl<K, V> implements TieredCache<K, V> {

    @Getter private final String name;
    private final String l2KeyPrefix;
    private final Duration l2Ttl;
    private final Function<K, V> loader;
    private final JavaType valueJavaType;
    private final ObjectMapper objectMapper;
    private final StringRedisTemplate redis;
    private final LoadingCache<K, Optional<V>> l1;

    // 失效广播 channel：tiered-cache:{name}:invalidate
    String invalidateChannel() { return "tiered-cache:" + name + ":invalidate"; }

    TieredCacheImpl(String name, int l1MaxSize, Duration l1RefreshAfterWrite,
                    String l2KeyPrefix, Duration l2Ttl, Function<K, V> loader,
                    JavaType valueJavaType, ObjectMapper objectMapper,
                    StringRedisTemplate redis) {
        this.name = name;
        this.l2KeyPrefix = l2KeyPrefix;
        this.l2Ttl = l2Ttl;
        this.loader = loader;
        this.valueJavaType = valueJavaType;
        this.objectMapper = objectMapper;
        this.redis = redis;

        // L1: refreshAfterWrite → 到期时先查 L2，L2 也过期才穿透 L3
        this.l1 = Caffeine.newBuilder()
                .maximumSize(l1MaxSize)
                .refreshAfterWrite(l1RefreshAfterWrite)
                .build(new CacheLoader<K, Optional<V>>() {
                    @Override
                    public Optional<V> load(K key) {
                        return loadFromL2OrL3(key);
                    }

                    @Override
                    public Optional<V> reload(K key, Optional<V> oldValue) {
                        // 续期逻辑：优先从 L2 取（不穿透 L3）
                        String json = redis.opsForValue().get(l2Key(key));
                        if (json != null) {
                            log.debug("[TIERED_CACHE][{}] L1 续期 from L2 key={}", name, key);
                            return Optional.ofNullable(deserialize(json));
                        }
                        // L2 也过期，穿透 L3
                        return loadFromL2OrL3(key);
                    }
                });
    }

    @Override
    public Optional<V> get(K key) {
        return l1.get(key);
    }

    @Override
    public void put(K key, V value) {
        l1.put(key, Optional.ofNullable(value));
        if (value != null) {
            redis.opsForValue().set(l2Key(key), serialize(value), l2Ttl);
        }
    }

    @Override
    public void invalidate(K key) {
        l1.invalidate(key);
        redis.delete(l2Key(key));
        // 广播给所有实例
        redis.convertAndSend(invalidateChannel(), keyToString(key));
        log.debug("[TIERED_CACHE][{}] invalidate key={}", name, key);
    }

    @Override
    public void refresh(K key) {
        l1.invalidate(key);
        redis.delete(l2Key(key));
        l1.get(key); // 触发重新加载
    }

    /** 收到 Pub/Sub 失效广播后调用（由 TieredCacheManager 回调） */
    void evictL1(String rawKey) {
        // rawKey 是 keyToString 序列化后的值，L1 key 是原始 K 类型
        // 直接清除整个 L1（保守策略）或解析 key（复杂）
        // 此处采用 invalidateAll 策略：广播时清所有 L1（安全，代价是 L3 穿透）
        // 更精确的方案：注册 K 的反序列化器（留后续扩展）
        l1.asMap().keySet().removeIf(k -> keyToString(k).equals(rawKey));
        log.debug("[TIERED_CACHE][{}] L1 evict from pub/sub rawKey={}", name, rawKey);
    }

    // ── 内部工具 ──────────────────────────────────────────────────

    private Optional<V> loadFromL2OrL3(K key) {
        // L2
        String json = redis.opsForValue().get(l2Key(key));
        if (json != null) {
            log.debug("[TIERED_CACHE][{}] L2 hit key={}", name, key);
            return Optional.ofNullable(deserialize(json));
        }
        // L3
        log.debug("[TIERED_CACHE][{}] L3 load key={}", name, key);
        V value = loader.apply(key);
        if (value != null) {
            redis.opsForValue().set(l2Key(key), serialize(value), l2Ttl);
        }
        return Optional.ofNullable(value);
    }

    private String l2Key(K key) { return l2KeyPrefix + keyToString(key); }
    private String keyToString(K key) { return key.toString(); }

    private String serialize(V value) {
        try { return objectMapper.writeValueAsString(value); }
        catch (Exception e) { throw new IllegalStateException("Cache serialize failed", e); }
    }

    private V deserialize(String json) {
        try { return objectMapper.readValue(json, valueJavaType); }
        catch (Exception e) {
            log.warn("[TIERED_CACHE][{}] deserialize failed, returning null: {}", name, e.getMessage());
            return null;
        }
    }
}
```

package org.chovy.cache;

import jakarta.annotation.PostConstruct;
import lombok.Getter;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.connection.ReactiveRedisConnectionFactory;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.data.redis.listener.PatternTopic;
import org.springframework.data.redis.listener.ReactiveRedisMessageListenerContainer;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * 多级缓存管理器。
 * 负责：1. 维护所有 TieredCache 实例注册表；2. 订阅 Redis Pub/Sub 失效广播。
 */
@Slf4j
@RequiredArgsConstructor
public class TieredCacheManager {

    @Getter
    private final StringRedisTemplate redis;
    private final ReactiveRedisConnectionFactory reactiveFactory;

    private final Map<String, TieredCacheImpl<?, ?>> caches = new ConcurrentHashMap<>();

    @PostConstruct
    void subscribeInvalidations() {
        try {
            ReactiveRedisMessageListenerContainer container =
                    new ReactiveRedisMessageListenerContainer(reactiveFactory);
            // 订阅所有缓存的失效通道：tiered-cache:*:invalidate
            container.receive(PatternTopic.of("tiered-cache:*:invalidate"))
                    .subscribe(
                            msg -> {
                                String channel = msg.getChannel();
                                String rawKey  = msg.getMessage();
                                // 从 channel 提取 cacheName：tiered-cache:{name}:invalidate
                                String[] parts = channel.split(":");
                                if (parts.length >= 3) {
                                    String cacheName = parts[1];
                                    TieredCacheImpl<?, ?> cache = caches.get(cacheName);
                                    if (cache != null) cache.evictL1(rawKey);
                                }
                            },
                            e -> log.error("[TIERED_CACHE_MANAGER] Pub/Sub error: {}", e.getMessage())
                    );
            log.info("[TIERED_CACHE_MANAGER] 已订阅 tiered-cache:*:invalidate");
        } catch (Exception e) {
            log.warn("[TIERED_CACHE_MANAGER] Pub/Sub 订阅失败（Redis 不可用时忽略）: {}", e.getMessage());
        }
    }

    /** 注册一个 TieredCache 实例（由 Builder.build() 调用） */
    void register(TieredCacheImpl<?, ?> cache) {
        caches.put(cache.getName(), cache);
        log.info("[TIERED_CACHE_MANAGER] 注册缓存 name={}", cache.getName());
    }
}
```

package org.chovy.cache;

import com.fasterxml.jackson.databind.ObjectMapper;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.data.redis.core.ValueOperations;

import java.time.Duration;
import java.util.Optional;
import java.util.concurrent.atomic.AtomicInteger;

import static org.assertj.core.api.Assertions.*;
import static org.mockito.ArgumentMatchers.*;
import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)
class TieredCacheTest {

    @Mock StringRedisTemplate redis;
    @Mock ValueOperations<String, String> valueOps;

    @BeforeEach
    void setUp() {
        when(redis.opsForValue()).thenReturn(valueOps);
    }

    private TieredCacheImpl<String, String> buildCache(java.util.function.Function<String, String> loader) {
        return new TieredCacheImpl<>(
                "test-cache", 100, Duration.ofSeconds(1),
                "test:", Duration.ofMinutes(10),
                loader, new ObjectMapper().constructType(String.class),
                new ObjectMapper(), redis
        );
    }

    @Test
    @DisplayName("L3 Loader 只在 L1+L2 都 miss 时被调用")
    void loader_called_only_on_full_miss() {
        AtomicInteger loaderCalls = new AtomicInteger(0);
        when(valueOps.get(anyString())).thenReturn(null); // L2 miss

        TieredCacheImpl<String, String> cache = buildCache(k -> {
            loaderCalls.incrementAndGet();
            return "value-" + k;
        });

        Optional<String> r1 = cache.get("k1");
        Optional<String> r2 = cache.get("k1"); // L1 hit, loader NOT called again

        assertThat(r1).contains("value-k1");
        assertThat(r2).contains("value-k1");
        assertThat(loaderCalls.get()).isEqualTo(1); // Loader 只调一次
    }

    @Test
    @DisplayName("invalidate 后再次 get 触发 Loader 重新加载")
    void invalidate_causes_reload() {
        AtomicInteger calls = new AtomicInteger(0);
        when(valueOps.get(anyString())).thenReturn(null);

        TieredCacheImpl<String, String> cache = buildCache(k -> {
            calls.incrementAndGet();
            return "v" + calls.get();
        });

        cache.get("k1"); // 第一次 → Loader
        cache.invalidate("k1");
        cache.get("k1"); // invalidate 后 → Loader 再次调用

        assertThat(calls.get()).isEqualTo(2);
    }

    @Test
    @DisplayName("L2 命中时不调用 Loader")
    void l2_hit_skips_loader() {
        AtomicInteger calls = new AtomicInteger(0);
        when(valueOps.get("test:l2key")).thenReturn("\"cached-value\""); // L2 命中

        TieredCacheImpl<String, String> cache = buildCache(k -> {
            calls.incrementAndGet();
            return "from-loader";
        });

        Optional<String> result = cache.get("l2key");

        assertThat(result).contains("cached-value");
        assertThat(calls.get()).isZero(); // Loader 未调用
    }
}
```

- [ ] **Step 1：创建 CacheConfig 配置类**

```java
package org.chovy.canvas.config;

import org.chovy.cache.TieredCache;
import org.chovy.cache.TieredCacheBuilder;
import org.chovy.cache.TieredCacheManager;
import org.chovy.canvas.domain.canvas.CanvasVersionMapper;
import org.chovy.canvas.engine.dag.DagGraph;
import org.chovy.canvas.engine.dag.DagParser;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.time.Duration;

@Configuration
public class CacheConfig {

    @Bean
    TieredCache<Long, DagGraph> canvasConfigCache(TieredCacheManager manager,
                                                   CanvasVersionMapper mapper,
                                                   DagParser parser) {
        return TieredCacheBuilder.<Long, DagGraph>builder()
                .name("canvas-config")
                .l1MaxSize(500)
                .l1RefreshAfterWrite(Duration.ofHours(2))
                .l2KeyPrefix("canvas:config:")
                .l2Ttl(Duration.ofHours(24))
                .loader(versionId -> {
                    var version = mapper.selectById(versionId);
                    if (version == null || version.getGraphJson() == null || version.getGraphJson().isBlank())
                        throw new IllegalStateException("版本 " + versionId + " 的 graph_json 不可用");
                    return parser.parse(version.getGraphJson());
                })
                .valueType(DagGraph.class)
                .build(manager);
    }
}
```

- [ ] **Step 2：重写 CanvasConfigCache 为薄包装层**

完整替换 `CanvasConfigCache.java`：

```java
package org.chovy.canvas.infra.cache;

import lombok.RequiredArgsConstructor;
import org.chovy.cache.TieredCache;
import org.chovy.canvas.engine.dag.DagGraph;
import org.springframework.stereotype.Service;

/**
 * 画布配置缓存——委托给 TieredCache SDK。
 * 所有生命周期管理（L1/L2/Pub/Sub）由 SDK 处理。
 */
@Service
@RequiredArgsConstructor
public class CanvasConfigCache {

    private final TieredCache<Long, DagGraph> canvasConfigCache;

    public DagGraph get(Long canvasId, Long versionId) {
        return canvasConfigCache.getOrThrow(versionId);
    }

    public void invalidate(Long canvasId, Long versionId) {
        canvasConfigCache.invalidate(versionId);
    }

    /** 供测试和外部调用直接清除 L1（不广播） */
    public void evictL1(String l1Key) {
        // SDK 的 invalidate 已包含 L1 eviction，直接调用
        // 此方法保留兼容性
    }
}
```
